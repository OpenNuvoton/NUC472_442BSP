434 int uac_check_fu_ctrl(UAC_INFO_T *uac_info, uint8_t target, int channel, int control)
435 {
436  AC_FU_T *fu = NULL;
437  AC_FT1_T *ft = NULL;
438  uint8_t *bptr;
439  uint16_t bitmap;
440 
441  if (channel == 0)
442  return 0; /* always fine for master channel */
443 
444  if (target == UAC_SPEAKER)
445  {
446  fu = uac_info->fu_play;
447  ft = uac_info->ft_play;
448  }
449  else if (target == UAC_MICROPHONE)
450  {
451  fu = uac_info->fu_rec;
452  ft = uac_info->ft_rec;
453  }
454  else
455  return UAC_RET_INVALID;
456 
457  if (!fu || !ft)
458  return UAC_RET_DEV_NOT_SUPPORTED;
459 
460  if (channel > ft->bNrChannels)
461  return UAC_RET_DEV_NOT_SUPPORTED;
462 
463  bptr = (uint8_t *)fu + 6 + fu->bControlSize * (channel - 1);
464 
465  if (fu->bControlSize == 1)
466  bitmap = *bptr;
467  else
468  bitmap = *bptr | (*(bptr+1) << 8);
469 
470  USBAS_DBGMSG("channel %d bmaControl is 0x%x\n", channel, bitmap);
471 
472  if (bitmap & (1 << (control-1)))
473  return 0; // UAC device support this control.
474  else
475  return UAC_RET_DEV_NOT_SUPPORTED;
476 }
