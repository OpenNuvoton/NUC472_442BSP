115 void USBD_ProcessSetupPacket(void)
116 {
117  // Setup packet process
118  gUsbCmd.bmRequestType = (uint8_t)(USBD->SETUP1_0 & 0xff);
119  gUsbCmd.bRequest = (int8_t)(USBD->SETUP1_0 >> 8) & 0xff;
120  gUsbCmd.wValue = (uint16_t)USBD->SETUP3_2;
121  gUsbCmd.wIndex = (uint16_t)USBD->SETUP5_4;
122  gUsbCmd.wLength = (uint16_t)USBD->SETUP7_6;
123 
124  /* USB device request in setup packet: offset 0, D[6..5]: 0=Standard, 1=Class, 2=Vendor, 3=Reserved */
125  switch (gUsbCmd.bmRequestType & 0x60)
126  {
127  case REQ_STANDARD: // Standard
128  {
129  USBD_StandardRequest();
130  break;
131  }
132  case REQ_CLASS: // Class
133  {
134  if (g_usbd_pfnClassRequest != NULL)
135  {
136  g_usbd_pfnClassRequest();
137  }

139  }

142  if (g_usbd_pfnVendorRequest != NULL)
143  {
144  g_usbd_pfnVendorRequest();
145  }
146  break;
147  }
148  default: // reserved
149  {
150  /* Setup error, stall the device */

178  u32Len = Minimum(u32Len, LEN_DEVICE);
179  USBD_PrepareCtrlIn((uint8_t *)g_usbd_sInfo->gu8DevDesc, u32Len);
180  break;
181  }

185  uint32_t u32TotalLen;
186

201  USBD_PrepareCtrlIn((uint8_t *)g_usbd_sInfo->gu8QualDesc, u32Len);

205  case DESC_OTHERSPEED:

245  g_usbd_CtrlZero = 1;

247  }

251  return 1;
252  }
253  break;
254  }

256  // Not support. Reply STALL.

273 void USBD_StandardRequest(void)
274 {
275  /* clear global variables for new request */
276  g_usbd_CtrlInPointer = 0;
277  g_usbd_CtrlInSize = 0;
278 
279  if (gUsbCmd.bmRequestType & 0x80) /* request data transfer direction */
280  {
281  // Device to host
282  switch (gUsbCmd.bRequest)
283  {
284  case GET_CONFIGURATION:
285  {
286  // Return current configuration setting
287  USBD_PrepareCtrlIn((uint8_t *)&g_usbd_UsbConfig, 1);
288 
289  USBD_CLR_CEP_INT_FLAG(USBD_CEPINTSTS_INTKIF_Msk);
290  USBD_ENABLE_CEP_INT(USBD_CEPINTEN_INTKIEN_Msk);
291  break;
292  }
293  case GET_DESCRIPTOR:
294  {
295  if (!USBD_GetDescriptor())
296  {
297  USBD_CLR_CEP_INT_FLAG(USBD_CEPINTSTS_INTKIF_Msk);
298  USBD_ENABLE_CEP_INT(USBD_CEPINTEN_INTKIEN_Msk);
299  }
300  break;
301  }
302  case GET_INTERFACE:
303  {
304  // Return current interface setting
305  USBD_PrepareCtrlIn((uint8_t *)&g_usbd_UsbAltInterface, 1);
306 
307  USBD_CLR_CEP_INT_FLAG(USBD_CEPINTSTS_INTKIF_Msk);
308  USBD_ENABLE_CEP_INT(USBD_CEPINTEN_INTKIEN_Msk);
309  break;
310  }
311  case GET_STATUS:
312  {
313  // Device
314  if (gUsbCmd.bmRequestType == 0x80)
315  {
316  if (g_usbd_sInfo->gu8ConfigDesc[7] & 0x40)
317  g_usbd_buf[0] = 1; // Self-Powered
318  else
319  g_usbd_buf[0] = 0; // bus-Powered
320  }
321  // Interface
322  else if (gUsbCmd.bmRequestType == 0x81)
323  g_usbd_buf[0] = 0;
324  // Endpoint
325  else if (gUsbCmd.bmRequestType == 0x82)
326  {
327  uint8_t ep = gUsbCmd.wIndex & 0xF;
328  g_usbd_buf[0] = USBD_GetStall(ep)? 1 : 0;
329  }
330  g_usbd_buf[1] = 0;
331  USBD_PrepareCtrlIn(g_usbd_buf, 2);
332  USBD_CLR_CEP_INT_FLAG(USBD_CEPINTSTS_INTKIF_Msk);
333  USBD_ENABLE_CEP_INT(USBD_CEPINTEN_INTKIEN_Msk);
334  break;
335  }
336  default:
337  {
338  /* Setup error, stall the device */
339  USBD_SET_CEP_STATE(USBD_CEPCTL_STALLEN_Msk);
340  break;
341  }
342  }
343  }
344  else
345  {
346  // Host to device
347  switch (gUsbCmd.bRequest)
348  {
349  case CLEAR_FEATURE:
350  {
351  if((gUsbCmd.wValue & 0xff) == FEATURE_ENDPOINT_HALT)
352  {
353 
354  int32_t epNum, i;
355 
356  /* EP number stall is not allow to be clear in MSC class "Error Recovery Test".
357  a flag: g_u32EpStallLock is added to support it */
358  epNum = gUsbCmd.wIndex & 0xF;
359  for (i=0; i<USBD_MAX_EP; i++)
360  {
361  if ((((USBD->EP[i].EPCFG & 0xf0) >> 4) == epNum) && ((g_u32EpStallLock & (1 << i)) == 0))
362  {
363  USBD->EP[i].EPRSPCTL = (USBD->EP[i].EPRSPCTL & 0xef) | USB_EP_RSPCTL_TOGGLE;
364  }
365  }
366  }
367  /* Status stage */
368  USBD_CLR_CEP_INT_FLAG(USBD_CEPINTSTS_STSDONEIF_Msk);
369  USBD_SET_CEP_STATE(USB_CEPCTL_NAKCLR);
370  USBD_ENABLE_CEP_INT(USBD_CEPINTEN_STSDONEIEN_Msk);
371  break;
372  }
373  case SET_ADDRESS:
374  {
375  g_usbd_UsbAddr = (uint8_t)gUsbCmd.wValue;
376 
377  // DATA IN for end of setup
378  /* Status Stage */
379  USBD_CLR_CEP_INT_FLAG(USBD_CEPINTSTS_STSDONEIF_Msk);
380  USBD_SET_CEP_STATE(USB_CEPCTL_NAKCLR);
381  USBD_ENABLE_CEP_INT(USBD_CEPINTEN_STSDONEIEN_Msk);
382  break;
383  }
384  case SET_CONFIGURATION:
385  {
386  g_usbd_UsbConfig = (uint8_t)gUsbCmd.wValue;
387  g_usbd_Configured = 1;
388  // DATA IN for end of setup
389  /* Status stage */
390  USBD_CLR_CEP_INT_FLAG(USBD_CEPINTSTS_STSDONEIF_Msk);
391  USBD_SET_CEP_STATE(USB_CEPCTL_NAKCLR);
392  USBD_ENABLE_CEP_INT(USBD_CEPINTEN_STSDONEIEN_Msk);
393  break;
394  }
395  case SET_FEATURE:
396  {
397  if ((gUsbCmd.wValue & 0x3) == 2) /* TEST_MODE*/
398  {
399  g_usbd_EnableTestMode = 1;
400  g_usbd_TestSelector = gUsbCmd.wIndex >> 8;
401  }
402  /* Status stage */
403  USBD_CLR_CEP_INT_FLAG(USBD_CEPINTSTS_STSDONEIF_Msk);
404  USBD_SET_CEP_STATE(USB_CEPCTL_NAKCLR);
405  USBD_ENABLE_CEP_INT(USBD_CEPINTEN_STSDONEIEN_Msk);
406  break;
407  }
408  case SET_INTERFACE:
409  {
410  g_usbd_UsbAltInterface = (uint8_t)gUsbCmd.wValue;
411  if (g_usbd_pfnSetInterface != NULL)
412  g_usbd_pfnSetInterface(g_usbd_UsbAltInterface);
413  /* Status stage */
414  USBD_CLR_CEP_INT_FLAG(USBD_CEPINTSTS_STSDONEIF_Msk);
415  USBD_SET_CEP_STATE(USB_CEPCTL_NAKCLR);
416  USBD_ENABLE_CEP_INT(USBD_CEPINTEN_STSDONEIEN_Msk);
417  break;
418  }
419  default:
420  {
421  /* Setup error, stall the device */
422  USBD_SET_CEP_STATE(USBD_CEPCTL_STALLEN_Msk);
423  break;
424  }
425  }

529 void USBD_CtrlIn(void)
530 {
531  int volatile i;
532  uint32_t volatile count;
533 
534  // Process remained data
535  if(g_usbd_CtrlInSize >= g_usbd_CtrlMaxPktSize)
536  {
537  // Data size > MXPLD
538  for (i=0; i<(g_usbd_CtrlMaxPktSize >> 2); i++, g_usbd_CtrlInPointer+=4)
539  USBD->CEPDAT = *(uint32_t *)g_usbd_CtrlInPointer;
540  USBD_START_CEP_IN(g_usbd_CtrlMaxPktSize);
541  g_usbd_CtrlInSize -= g_usbd_CtrlMaxPktSize;
542  }
543  else
544  {
545  // Data size <= MXPLD
546  for (i=0; i<(g_usbd_CtrlInSize >> 2); i++, g_usbd_CtrlInPointer+=4)
547  USBD->CEPDAT = *(uint32_t *)g_usbd_CtrlInPointer;
548 
549  count = g_usbd_CtrlInSize % 4;
550  for (i=0; i<count; i++)
551  USBD->CEPDAT_BYTE = *(uint8_t *)(g_usbd_CtrlInPointer + i);
552 
553  USBD_START_CEP_IN(g_usbd_CtrlInSize);
554  g_usbd_CtrlInPointer = 0;
555  g_usbd_CtrlInSize = 0;
556  }
