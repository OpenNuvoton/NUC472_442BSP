266 uint32_t FMC_CRC8(uint32_t au32Data[], int i32Count)
267 {
268  int i32ByteIdx;
269  uint8_t i, u8Cnt, u8InData;
270  uint8_t au8CRC[4] = { 0xff, 0xff, 0xff, 0xff };
271 
272  for (i32ByteIdx = 0; i32ByteIdx < 4; i32ByteIdx++)
273  {
274  for (u8Cnt = 0; u8Cnt < i32Count; u8Cnt++)
275  {
276  for (i = 0x80; i != 0; i /= 2)
277  {
278  if ((au8CRC[i32ByteIdx] & 0x80)!=0)
279  {
280  au8CRC[i32ByteIdx] *= 2;
281  au8CRC[i32ByteIdx] ^= 7;
282  }
283  else
284  au8CRC[i32ByteIdx] *= 2;
285 
286  u8InData = (au32Data[u8Cnt] >> (i32ByteIdx * 8)) & 0xff;
287 
288  if ((u8InData & i) != 0)
289  au8CRC[i32ByteIdx]^=0x7;
290  }
291  }
292  }
293  return (au8CRC[0] | au8CRC[1] << 8 | au8CRC[2] << 16 | au8CRC[3] << 24);
294 }
