172  return 0; /* PLL is in power down mode or fix low */

183  u32NO=1;

662  if(u32PllClkSrc == CLK_PLLCTL_PLLSRC_HXT)

673

675  u32NR = 2;
676  }
677

679  else
680  {
681  /* Enable HIRC clock */
682  CLK->PWRCTL |= CLK_PWRCTL_HIRCEN_Msk;
683 
684  /* Wait for HIRC clock ready */
685  CLK_WaitClockReady(CLK_STATUS_HIRCSTB_Msk);
686

688  u32CLK_SRC = CLK_PLLCTL_PLLSRC_HIRC;
689  u32PllSrcClk = __HIRC;
690 
691  /* u32NR start from 4 when FIN = 22.1184MHz to avoid calculation overflow */
692  u32NR = 4;
693  }
694 
695  /* Select "NO" according to request frequency */
696  if((u32PllFreq <= FREQ_500MHZ) && (u32PllFreq > FREQ_250MHZ))
697  {
698  u32NO = 0;
699  }
700  else if((u32PllFreq <= FREQ_250MHZ) && (u32PllFreq > FREQ_125MHZ))
701  {
702  u32NO = 1;
703  u32PllFreq = u32PllFreq << 1;
704  }
705  else if((u32PllFreq <= FREQ_125MHZ) && (u32PllFreq >= FREQ_50MHZ))
706  {
707  u32NO = 3;
708  u32PllFreq = u32PllFreq << 2;
709  }
710  else
711  {
712  /* Wrong frequency request. Just return default setting. */
713  goto lexit;
714  }
715 
716  /* Find best solution */
717  u32Min = (uint32_t) - 1;
718  u32MinNR = 0;
719  u32MinNF = 0;
720  for(; u32NR <= 33; u32NR++)
721  {
722  u32Tmp = u32PllSrcClk / u32NR;
723  if((u32Tmp > 1600000) && (u32Tmp < 16000000))
724  {
725  for(u32NF = 2; u32NF <= 513; u32NF++)
726  {
727  u32Tmp2 = u32Tmp * u32NF;
728  if((u32Tmp2 >= 200000000) && (u32Tmp2 <= 500000000))
729  {
730  u32Tmp3 = (u32Tmp2 > u32PllFreq) ? u32Tmp2 - u32PllFreq : u32PllFreq - u32Tmp2;
731  if(u32Tmp3 < u32Min)
732  {
733  u32Min = u32Tmp3;
734  u32MinNR = u32NR;
735  u32MinNF = u32NF;
736 
737  /* Break when get good results */
738  if(u32Min == 0)
739  break;
740  }
741  }
742  }
743  }
744  }
745 
746  /* Enable and apply new PLL setting. */
747  CLK->PLLCTL = u32CLK_SRC | (u32NO << 14) | ((u32MinNR - 2) << 9) | (u32MinNF - 2);
748 
749  /* Wait for PLL clock stable */
750  CLK_WaitClockReady(CLK_STATUS_PLLSTB_Msk);
751 
752  /* Return actual PLL output clock frequency */
753  return u32PllSrcClk / ((u32NO + 1) * u32MinNR) * u32MinNF;
754

762

764  CLK_WaitClockReady(CLK_STATUS_PLLSTB_Msk);
765 
766  return CLK_GetPLLClockFreq();
