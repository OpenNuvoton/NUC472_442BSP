176 {

182  /*------------------------------------------------------------------------------------*/

184  td_fill(td_idx, info, (uint8_t *)&_request, 8);
185  td_idx++;
186 
187  /*------------------------------------------------------------------------------------*/

189  /*------------------------------------------------------------------------------------*/
190  if (data_len > 0)
191  {
192  if (pipe_out)
193  info = (TD_CC | TD_R | TD_DP_OUT | TD_T_DATA1);
194  else
195  info = (TD_CC | TD_R | TD_DP_IN | TD_T_DATA1);
196 
197  td_fill(td_idx, info, data_buff, data_len);
198  td_idx++;
199  }
200 
201  /*------------------------------------------------------------------------------------*/
202  /* prepare STATUS stage TD */
203  /*------------------------------------------------------------------------------------*/
204  if (pipe_out)
205  info = (TD_CC | TD_DP_IN | TD_T_DATA1);
206  else
207  info = (TD_CC | TD_DP_OUT | TD_T_DATA1);
208

210  _last_td_idx = td_idx;
211 
212  td_idx++;
213

216  _td[td_idx].hwNextTD = 0;;
217

235  uint8_t *data_buff, int data_len, int pipe_out)
236 {

244  info = (TD_CC | TD_R | TD_DP_OUT);
245  else
246  info = (TD_CC | TD_R | TD_DP_IN);
247

276  | (0 << 11) /* Direction */
