137 void SC_ResetReader(SC_T *sc)
138 {
139  uint32_t u32Intf;
140 
141  if(sc == SC0)
142  u32Intf = 0;
143  else if(sc == SC1)
144  u32Intf = 1;
145  else if(sc == SC2)
146  u32Intf = 2;
147  else if(sc == SC3)
148  u32Intf = 3;
149  else if(sc == SC4)
150  u32Intf = 4;
151  else
152  u32Intf = 5;
153 
154  // Reset FIFO, enable auto de-activation while card removal
155  sc->ALTCTL |= (SC_ALTCTL_TXRST_Msk | SC_ALTCTL_RXRST_Msk | SC_ALTCTL_ADACEN_Msk);
156  // Set Rx trigger level to 1 character, longest card detect debounce period, disable error retry (EMV ATR does not use error retry)
157  while(sc->CTL & SC_CTL_SYNC_Msk);
158  sc->CTL &= ~(SC_CTL_RXTRGLV_Msk |
159  SC_CTL_CDDBSEL_Msk |
160  SC_CTL_TXRTY_Msk |
161  SC_CTL_TXRTYEN_Msk |
162  SC_CTL_RXRTY_Msk |
163  SC_CTL_RXRTYEN_Msk);
164  while(sc->CTL & SC_CTL_SYNC_Msk);
165  // Enable auto convention, and all three smartcard internal timers
166  sc->CTL |= SC_CTL_AUTOCEN_Msk | SC_CTL_TMRSEL_Msk;
167  // Disable Rx timeout
168  sc->RXTOUT = 0;
169  // 372 clocks per ETU by default
170  sc->ETUCTL= 371;
171 
172 
173  /* Enable necessary interrupt for smartcard operation */
174  if(u32CardStateIgnore[u32Intf]) // Do not enable card detect interrupt if card present state ignore
175  sc->INTEN = (SC_INTEN_RDAIEN_Msk |
176  SC_INTEN_TERRIEN_Msk |
177  SC_INTEN_TMR0IEN_Msk |
178  SC_INTEN_TMR1IEN_Msk |
179  SC_INTEN_TMR2IEN_Msk |
180  SC_INTEN_BGTIEN_Msk |
181  SC_INTEN_ACERRIEN_Msk);
182  else
183  sc->INTEN = (SC_INTEN_RDAIEN_Msk |
184  SC_INTEN_TERRIEN_Msk |
185  SC_INTEN_TMR0IEN_Msk |
186  SC_INTEN_TMR1IEN_Msk |
187  SC_INTEN_TMR2IEN_Msk |
188  SC_INTEN_BGTIEN_Msk |
189  SC_INTEN_CDIEN_Msk |
190  SC_INTEN_ACERRIEN_Msk);
191 
192  return;
193 }
