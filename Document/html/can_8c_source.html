432 static int can_update_spt(int sampl_pt, int tseg, int *tseg1, int *tseg2)
433 {

459 uint32_t CAN_SetBaudRate(CAN_T *tCAN, uint32_t u32BaudRate)
460 {
461  long rate;
462  long best_error = 1000000000, error = 0;
463  int best_tseg = 0, best_brp = 0, brp = 0;
464  int tsegall, tseg = 0, tseg1 = 0, tseg2 = 0;
465  int spt_error = 1000, spt = 0, sampl_pt;
466  uint64_t clock_freq = 0;
467  uint32_t sjw = 1;
468 
469  CAN_EnterInitMode(tCAN);
470 
471  SystemCoreClockUpdate();
472 
473  clock_freq = CLK_GetPCLKFreq();
474 
475  if(u32BaudRate >= 1000000)
476  u32BaudRate = 1000000;
477 
478  /* Use CIA recommended sample points */
479  if (u32BaudRate > 800000)
480  sampl_pt = 750;
481  else if (u32BaudRate > 500000)
482  sampl_pt = 800;
483  else
484  sampl_pt = 875;
485 
486  /* tseg even = round down, odd = round up */
487  for (tseg = (TSEG1_MAX + TSEG2_MAX) * 2 + 1; tseg >= (TSEG1_MIN + TSEG2_MIN) * 2; tseg--)
488  {
489  tsegall = 1 + tseg / 2;
490  /* Compute all possible tseg choices (tseg=tseg1+tseg2) */
491  brp = clock_freq / (tsegall * u32BaudRate) + tseg % 2;
492  /* chose brp step which is possible in system */
493  brp = (brp / BRP_INC) * BRP_INC;
494 
495  if ((brp < BRP_MIN) || (brp > BRP_MAX))
496  continue;
497  rate = clock_freq / (brp * tsegall);
498 
499  error = u32BaudRate - rate;
500 
501  /* tseg brp biterror */
502  if (error < 0)
503  error = -error;
504  if (error > best_error)
505  continue;
506  best_error = error;
507  if (error == 0)
508  {
509  spt = can_update_spt(sampl_pt, tseg / 2, &tseg1, &tseg2);
510  error = sampl_pt - spt;
511  if (error < 0)
512  error = -error;
513  if (error > spt_error)
514  continue;
515  spt_error = error;
516  }
517  best_tseg = tseg / 2;
518  best_brp = brp;
519 
520  if (error == 0)
521  break;
522  }
523 
524  spt = can_update_spt(sampl_pt, best_tseg, &tseg1, &tseg2);
525 
526  /* check for sjw user settings */
527  /* bt->sjw is at least 1 -> sanitize upper bound to sjw_max */
528  if (sjw > SJW_MAX)
529  sjw = SJW_MAX;
530  /* bt->sjw must not be higher than tseg2 */
531  if (tseg2 < sjw)
532  sjw = tseg2;
533 
534  /* real bit-rate */
535  u32BaudRate = clock_freq / (best_brp * (tseg1 + tseg2 + 1));
536 
537  tCAN->BTIME = ((uint32_t)(tseg2 - 1) << CAN_BTIME_TSEG2_Pos) | ((uint32_t)(tseg1 - 1) << CAN_BTIME_TSEG1_Pos) |
538  ((best_brp - 1) & CAN_BTIME_BRP_Msk) | (sjw << CAN_BTIME_SJW_Pos);
539  tCAN->BRPE = ((best_brp - 1) >> 6) & 0x0F;
540 
541  //printf("\n bitrate = %d \n", CAN_GetCANBitRate(tCAN));
542 
543  CAN_LeaveInitMode(tCAN);
544 
545  return u32BaudRate;
546 }
